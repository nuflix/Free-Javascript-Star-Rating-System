Introduction:

To use rating system you just have to make div with class name "starRatingContainer" and make another div element inside it with class name of your choice: 
<div class="starRatingContainer"><div class="anyClassName"></div></div>
and call this function: 
rateSystem(string, arrayOfJsonObjects, function(rating, triggeredElement){});
  • First argument is a string. It is class name of div elements that will load your star rating.
  • Second argument is array of JSON objects. Each JSON object sets all properties of a div that has class name equal to the first argument.
  • last argument(function) is optional(it is triggered when user rates). It should have 2 parameters. First one is rating and the second one is element that has been triggered.

Examples:

html:


        //example 1
        <div>  <div class="starRatingContainer"> <div class="className"> </div> </div>  <div class="ratingHolder"> </div>  </div>
    
        //example 2
        <div> <div class="starRatingContainer"> <div class="ratingSystem"> </div> </div>  <div class="ratingHolder"> </div> </div>
       
        //example 3
        <div>  <div class="starRatingContainer"> <div class="className"> </div> </div>  <div class="ratingHolder"> </div>  </div>
       
        //example 4
        <div> <div class="starRatingContainer"> <div class="ratingSystem"> </div> </div>  <div class="ratingHolder"> </div> </div>
       
        //example 5
        <div>  <div class="starRatingContainer"> <div class="className"> </div> </div>  <div class="ratingHolder"> </div>  </div>

javaScript:


    <script>

    //setting variables
    var properties1=[
        {"rating":"4.3", "maxRating":"5", "readOnly":"no", "starImage":"./star.png", "emptyStarImage":"./starbackground.png", "starSize":"16", "step":"0.5"},
        {"rating":"2.7", "maxRating":"5", "readOnly":"yes", "starImage":"./star.png", "emptyStarImage":"./starbackground.png", "starSize":"32"}
    ];
    var properties2=[
        {"rating":"4.3", "maxRating":"5", "readOnly":"no", "starImage":"./star.png", "starSize":"16", "step":"0.5"},
        {"rating":"4.3", "maxRating":"10", "readOnly":"no", "starImage":"./star.png", "emptyStarImage":"./starbackground.png", "starSize":"16", "step":"1"},
        {"rating":"4.3", "maxRating":"5", "readOnly":"no", "starImage":"./star.png", "backgroundStarImage":"./backgroundStar.png", "starSize":"16", "step":"0.5"}
    ];
    var className="ratingSystem";

    //execution
    rateSystem(className, properties1, function(rating, ratingTargetElement){  ratingTargetElement.parentElement.parentElement.getElementsByClassName("ratingHolder")[0].innerHTML = rating; });

    rateSystem("className", properties2, function(rating, ratingTargetElement){  ratingTargetElement.parentElement.parentElement.getElementsByClassName("ratingHolder")[0].innerHTML = rating;  });
    
    </script>
